# 目录结构
```
apps                              # 各个商城门户项目主目录
├─b2b                             # 联营商城（B/C端企业商城）
├──.editorconfig
├──.gitignore
├──.npmrc
├──bootstrap.js
├──package.json
├──production.build.js
├──README.md
├──tsconfig.json
├──tslint.json
├──ytt.config.ts                  # yapi配置文件
├──web                            # 客户端页面目录
├──src                            # 服务端代码目录
├──output
├──config                         # 项目配置文件
├──build
├─channel                         # 渠道商城
├─info                            # 资讯门户
├─logistics                       # 物流门户
├─manufacture                     # 加工门户
├─platform                        # 平台首页
├─self                            # 自营商城
├─srm                             # 自营门户
locales                           # 国际化文件目录
packages                          # 商城公共方法/组件库
scripts                           # 脚本文件目录
.gitignore
.npmrc
README.md
env.js
lerna.json
package.json
```
